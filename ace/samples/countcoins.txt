//http://rosettacode.org/wiki/Count_the_coins#Dyalect
func countCoins(coins, n) {
    let xs = Array.Empty(n + 1, 0)
    xs[0] = 1
    for c in coins {
        var cj = c
        while cj <= n {
            xs[cj] += xs[cj - c]
            cj += 1
        }
    }
    xs[n]
}
 
let coins = [1, 5, 10, 25]
print(countCoins(coins, 100))
