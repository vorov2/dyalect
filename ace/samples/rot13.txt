//http://rosettacode.org/wiki/Rot-13#Dyalect
func Char.Rot13() {
    match this {
        (>= 'a' and <= 'm') or (>= 'A' and <= 'M') => Char(this.Order() + 13),
        (>= 'n' and <= 'z') or (>= 'N' and <= 'Z') => Char(this.Order() - 13),
        _ => this
    }
}
 
func String.Rot13() {
    var cs = []
    for c in this {
        cs.Add(c.Rot13())
    }
    String.Concat(values: cs)
}
 
"ABJURER nowhere".Rot13()
