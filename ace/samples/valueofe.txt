//http://rosettacode.org/wiki/Calculating_the_value_of_e#Dyalect
func calculateE(epsilon = 1.0e-15) {
    var fact = 1
    var e = 2.0
    var e0 = 0.0
    var n = 2
 
    while true {
        e0 = e
        fact *= n
        n += 1
        e += 1.0 / Float(fact)
 
        if abs(e - e0) < epsilon {
            break
        }
    }
 
    return e
}
 
print(calculateE())
